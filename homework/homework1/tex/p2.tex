\problem{}

Sort the following functions in ascending order of growth.
\begin{align}
    &f_1(n)=\log_2{n} \\
    &f_2(n)=n^{\sqrt{n}} \\
    &f_3(n)=(\log_2{n})^{\log_2{n}} \\
    &f_4(n)=n^{\frac{3}{4}}  \\
    &f_5(n)=\log_2{\log_2{n}} \\
    &f_6(n)=2^{\log_2{n}}  \\
    &f_7(n)=10^{2024}  \\
    &f_8(n)=5^n \\
    &f_9(n)=\log_2{n}\cdot\log_2{3^n} \\
    &f_{10}(n)=(\log_2{n})^n
\end{align}

\solution{}


$$f_7<f_5<f_1<f_4<f_6<f_9<f_3<f_8<f_{10}<f_2$$
